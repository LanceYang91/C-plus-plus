//function to order numbers in string
bool compareString(const string&a, const string&b){
	//for first string
	if (a.empty())
		return true;

	//for second string
	if (b.empty())
		return false;

	//if have a number
	if (isdigit(a[0]) && !isdigit(b[0]))
		return true;

	//if no number
	if (!isdigit(a[0]) && !isdigit(b[0]))
		return false;

	//if the same 
	if (!isdigit(a[0]) && !isdigit(b[0])){
		if (a[0] == b[0])
			return compareString(a.substr(1), b.substr(1));
			return (toupper(a[0]) < toupper(b[0]));
	}

	//both string begin with digit, analyze numbers
	istringstream issa(a);
	istringstream issb(b);
	int ia, ib;
	issa >> ia;
	issb >> ib;

	if (ia != ib)
		return ia < ib;

	//If numbers are the same
	string anew, bnew;
	getline (issa, anew);
	getline (issb, bnew);
	return (compareString(anew, bnew));

}
